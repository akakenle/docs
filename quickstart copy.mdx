## 直接调用 aihubmix API

其中 "AIHUBMIX_API_KEY" 替换为 [aihubmix key](https://aihubmix.com/token)，注意 `key` 的有效期和额度限制\
可使用的 "model" 列表，可查阅 [模型广场](https://aihubmix.com/models) 复制模型名称替换即可

<Tabs>
<TabItem value="sh" label="Shell">

```sh
curl 'https://aihubmix.com/v1/chat/completions'
-H 'Authorization: Bearer <AIHUBMIX_API_KEY>'
-H 'Content-Type: application/json'
-d '{
   "model": "gpt-3.5-turbo",
   "messages": [
      {
         "role": "user",
         "content": "What is the meaning of life?"
      }
   ]
}'
```

</TabItem>
<TabItem value="js" label="JavaScript">

```js
// 请在https://aihubmix.com域名下尝试，否则有浏览器跨域问题
fetch('https://aihubmix.com/v1/chat/completions', {
  method: 'POST',
  headers: {
    Authorization: 'Bearer <AIHUBMIX_API_KEY>',
    'Content-Type': 'application/json',
  },
  body: JSON.stringify({
    model: "gpt-3.5-turbo",
    messages: [
      {
         "role": "user",
         "content": "What is the meaning of life?"
      }
    ]
  }),
})
```

</TabItem>
<TabItem value="py" label="Python">

```py
import requests
import json

response = requests.post(
  url="https://aihubmix.com/v1/chat/completions",
  headers={
    "Authorization": "Bearer <AIHUBMIX_API_KEY>",
    "Content-Type": "application/json",
  },
  data=json.dumps({
    "model": "gpt-3.5-turbo", # Optional
    "messages": [
      {
        "role": "user",
        "content": "What is the meaning of life?"
      }
    ]
  })
)

```

</TabItem>
</Tabs>


API 同样也支持流式数据, 参数增加 `stream: true`









## 使用 OpenAI SDK

其中 "AIHUBMIX_API_KEY" 替换为 [aihubmix key](https://aihubmix.com/token)，注意 `key` 的有效期和额度限制\
可使用的 "model" 列表，可查阅 [模型广场](https://aihubmix.com/models) 复制模型名称替换即可

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

<Tabs>
<TabItem value="js" label="JavaScript">

```js
import OpenAI from 'openai';

const openai = new OpenAI({
  baseURL: 'https://aihubmix.com/v1',
  apiKey: '<AIHUBMIX_API_KEY>',
});

async function main() {
  const completion = await openai.chat.completions.create({
    model: "gpt-3.5-turbo",
    messages: [
      {
         "role": "user",
         "content": "What is the meaning of life?"
      }
    ]
  });

  console.log(completion.choices[0].message);
}

main();

```

</TabItem>
<TabItem value="py" label="Python">

```py
from openai import OpenAI

client = OpenAI(
  base_url="https://aihubmix.com/v1",
  api_key="<AIHUBMIX_API_KEY>",
)

completion = client.chat.completions.create(
  model="gpt-3.5-turbo",
  messages=[
    {
      "role": "user",
      "content": "What is the meaning of life?"
    }
  ]
)

print(completion.choices[0].message.content)
```

</TabItem>
</Tabs>
